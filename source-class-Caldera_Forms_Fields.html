 14     /**
 15      * Get all field definitions
 16      *
 17      * @since 1.5.0
 18      *
 19      * @return array
 20      */
 21     public static function get_all() {
 22 
 23         /**
 24          * Register or remove field types
 25          *
 26          * @since unknown
 27          *
 28          * @param array $field_types Field types
 29          */
 30         $field_types = apply_filters( 'caldera_forms_get_field_types', self::internal_types() );
 31 
 32 
 33         if ( ! empty( $field_types ) ) {
 34             foreach ( $field_types as $fieldType => $fieldConfig ) {
 35                 // check for a viewer
 36                 if ( isset( $fieldConfig[ 'viewer' ] ) ) {
 37                     add_filter( 'caldera_forms_view_field_' . $fieldType, $fieldConfig[ 'viewer' ], 10, 3 );
 38                 }
 39             }
 40         }
 41 
 42         return $field_types;
 43 
 44     }

 46     /**
 47      * Get definition of one field
 48      *
 49      * @since 1.5.0
 50      *
 51      * @param string $type Field type
 52      *
 53      * @return array
 54      */
 55     public static function definition( $type ){
 56         $fields = self::get_all();
 57         if( array_key_exists( $type, $fields ) ){
 58             return $fields[ $type ];
 59         }
 60 
 61         return array();
 62 
 63     }

 65     /**
 66      * Check if a field definition has defined a specific "not support" argument
 67      *
 68      * @since 1.5.0
 69      *
 70      * @param string $type The field type
 71      * @param string $not_support The not support argument, for example "entry_list"
 72      *
 73      * @return bool|null True if not supported, false if not not supported. Null if invalid field type
 74      */
 75     public static function not_support( $type, $not_support ){
 76         $field = self::definition( $type );
 77         if( ! empty( $field ) ){
 78             if( ! isset( $field[ 'setup' ], $field[ 'setup' ][ 'not_supported' ] )  ){
 79                 return false;
 80             }
 81             if( ! empty(  $field[ 'setup' ][ 'not_supported' ] ) &&  in_array( $not_support, $field[ 'setup' ][ 'not_supported' ] )  ){
 82                 return true;
 83             }
 84 
 85             return false;
 86         }
 87 
 88         return null;
 89 
 90     }
