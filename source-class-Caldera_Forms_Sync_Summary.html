32          * Element type to wrap Caldera Forms summary fields in
33          *
34          * @since 1.5.0

36          * @param string $type Element type ONLY. For example 'div' not '<div>'.
37          * @param array $field Field config
38          * @param array $form Form config

41 
42         /**

44          *
45          * MUST have 2 %s
46          *
47          * @since 1.5.0
48          *
49          * @param string $summary_pattern The sprintf pattern for each item. With 2 %s -- no more, no less or errors will happen
50          * @param array $field Field config
51          * @param array $form Form config
52          */
53         $summary_pattern = apply_filters( 'caldera_forms_summary_field_pattern', '<li><span class="caldera-forms-summary-label">%s</span> <span class="caldera-forms-summary-value">%s</span></li>', $this->field, $this->form );
54 
55         $summary = '<' . $summary_wrap .' >';
56         foreach ( Caldera_Forms_Forms::get_fields( $this->form, true ) as $_field ){
57             if( $_field[ 'ID' ] == $this->field[ 'ID' ] || true === $this->dont_sync( Caldera_Forms_Field_Util::get_type( $_field, $this->form ) ) )  {
58                 continue;
59             }
60 
61             $magic_tag = '%' . $_field[ 'slug' ] . '%';
62             $summary .= sprintf( $summary_pattern, esc_html( $_field[ 'label' ] ), $magic_tag );
63         }
64 
65         $summary .= '</' . $summary_wrap . '>';
66         return $summary;
67     }
