 59      * @since 1.5.0

 61     public function prepare_data(){
 62 
 63         if( ! empty( $this->form[ 'fields' ] ) ){
 64             foreach( $this->form[ 'fields' ] as $field ){
 65                 $type = Caldera_Forms_Field_Util::get_type( $field, $this->form );
 66                 if( 'summary' == $type ){
 67                     $type = 'html';
 68                 }
 69                 //skip these types -- maybe add filter here later
 70                 $skip = array(
 71                     'calculation',
 72                     'star_rating',
 73                 );
 74                 if( ! in_array( $type, $skip ) && method_exists( $this, $type ) ){
 75                     call_user_func( array( $this, $type ), $field[ 'ID' ], $field );
 76                 }
 77             }
 78         }
 79     }
